206 void skip_comments() {
216  // Cria uma variável para armazenar cada caracter do comentário
217  char buffer = fgetc(file_img);
218 
219  // Repete o laço enquanto a variável de buffer for igual a um '#'
220  do {
221  if (buffer == '#')
222  // Se buffer for igual a '#', é por que está iniciando
223  // um comentário, então a variável buffer recebe cada
224  // caractere até que ele seja uma quebra de linha
225  while (buffer != '\n')
226  buffer = fgetc(file_img);
227  else
228  // Caso contrário, não é um comentário, então "devolve"
229  // para o arquivo, para que ele possa ser lido novamente
230  ungetc(buffer, file_img);
231  // buffer pega novamente um caracter (para verificar
232  // se tem mais algum comentário no arquivo)
233  buffer = fgetc(file_img);
234  } while (buffer == '#');
235  // Quando o laço acabar (o último caracter pego for
236  // diferente de um '#'), "devolve" para o arquivo o
237  // caracter que está em buffer, pois não é um comentário
238  ungetc(buffer, file_img);
239 }
